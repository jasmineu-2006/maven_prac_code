pipeline {
    agent any

    tools {
        maven 'M3'
    }

    stages {

        stage('Checkout Git') {
            steps {
                git branch: 'main',
                url: 'https://github.com/jasmineu-2006/maven_prac_code.git'
            }
        }

        stage('Build and Test') {
            steps {
                // For Windows use 'bat', for Linux use 'sh'
                bat 'mvn clean test'
            }
        }

        stage('Package') {
            steps {
                bat 'mvn package'
            }
        }
    }

    post {
        success {
            echo 'Build Successful!'
        }
        failure {
            echo 'Build Failed!'
        }
    }
}
